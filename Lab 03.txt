{   float a, b, c, delta, x1, x2;
    printf("Lembrando que: ax2 + bx + c = 0\nDigite os termos 'a', 'b' e 'c': ");
    scanf("%f %f %f", &a, &b, &c);
    delta= b*b - 4*a*c;
    if(a>0){
        if(delta>0){
                    x1=(-b+sqrt(delta))/2*a;
                    x2=(-b-sqrt(delta))/2*a;
                    printf("As raizes sao: %f %f", x1, x2);
        }
        if(delta==0){
                    x1=(-b+sqrt(delta))/2*a;
                    printf("Raiz unica: %f", x1);
        }
        if(delta<0){
                    printf("Nao existe raiz");
        }
    }else{
        printf("Nao eh equacao de segundo grau");
    }
    return 0;
}
